```dockerfile
# Dockerfile
# Build stage
FROM golang:1.21-alpine AS builder

RUN apk add --no-cache git ca-certificates

WORKDIR /app

COPY go.mod go.sum ./
RUN go mod download

COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -a -installsuffix cgo -o harbor-replicator ./cmd/replicator

# Final stage
FROM alpine:3.19

RUN apk --no-cache add ca-certificates tzdata

WORKDIR /app

COPY --from=builder /app/harbor-replicator .
COPY config/replicator.yaml /config/replicator.yaml

EXPOSE 8080 9090

USER 1000:1000

ENTRYPOINT ["./harbor-replicator"]
CMD ["--config", "/config/replicator.yaml"]
```
